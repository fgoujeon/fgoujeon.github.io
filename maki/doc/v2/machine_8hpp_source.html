481 template<detail::machine_operation Operation, class Event>
482 void execute_operation_now(const Event& event)
483 {
484 if constexpr(detail::impl_of(conf).run_to_completion)
485 {
486 auto grd = executing_operation_guard{*this};
487
488 execute_one_operation<Operation>(event);
489
490 /*
491 Process enqueued and deferred events, if any.
492 We guarantee order of processing: At any given state configuration,
493 if several pending events can be processed, they're processed in the
494 same order they've been given to the `machine`.
495 */
496 try_processing_deferred_operations();
497 while (!rtc_queue_.empty())
498 {
499 rtc_queue_.invoke_and_pop(*this);
500 try_processing_deferred_operations();
501 }
502 }
503 else
504 {
505 execute_one_operation<Operation>(event);
506
507 try_processing_deferred_operations();
508 }
509 }

524 /*
525 Process all previously deferred events that can now be processed.
526 */
527 void try_processing_deferred_operations()
528 {
529 if constexpr(has_deferrable_events)
530 {
531 /*
532 The inner loop tries to process every deferred event once.
533
534 The outer loop executes the inner loop as many times as necessary, that
535 is, until `event_deferral_queue_` only contains events that are still
536 deferred by any of the currently active states.
537
538 These two levels are necessary, as processing a previously deferred
539 event can change the active states and allow other events of
540 `event_deferral_queue_` to be processed.
541 */
542
543 auto processing_count = 1;
544 while (processing_count != 0) // Outer loop
545 {
546 processing_count = 0;
547 for (auto i = 0U; i < event_deferral_queue_.size(); ++i) // Inner loop
548 {
549 processing_count += static_cast<int>(event_deferral_queue_.invoke_and_pop(*this));
550 }
551 }
552 }
553 }

555 template<detail::machine_operation Operation, class Event>
556 bool execute_one_operation(const Event& event)
557 {
558 if constexpr(Operation == detail::machine_operation::start)
559 {
560 impl_.enter(*this, context(), event);
561 return true;
562 }
563 else if constexpr(Operation == detail::machine_operation::stop)
564 {
565 impl_.exit_to_finals(*this, context(), event);
566 return true;
567 }
568 else
569 {
570 constexpr auto is_deferrable_event = detail::type_set_contains_v
571 <
572 deferrable_event_type_set,
573 Event
574 >;
575
576 constexpr auto has_matching_pre_processing_hook = detail::tlu::contains_if_v
577 <
578 pre_processing_hook_ptr_constant_list,
579 detail::event_action_traits::for_event<Event>::template has_containing_event_set
580 >;
581
582 constexpr auto has_matching_post_processing_hook = detail::tlu::contains_if_v
583 <
584 post_processing_hook_ptr_constant_list,
585 detail::event_action_traits::for_event<Event>::template has_containing_event_set
586 >;
587
588 // Defer the event if required by any of the active states
589 if constexpr(is_deferrable_event)
590 {
591 if(impl_.template defers_event<Event>())
592 {
593 event_deferral_queue_.template push<any_event_visitor<Operation>>(event);
594 return false;
595 }
596 }
597
598 //If running, execute pre-processing hook for `Event`, if any.
599 if constexpr(has_matching_pre_processing_hook)
600 {
601 if(running())
602 {
603 detail::call_matching_event_action<pre_processing_hook_ptr_constant_list>
604 (
605 *this,
606 context(),
607 event
608 );
609 }
610 }
611
612 /*
613 If running:
614 - process the event;
615 - execute the post-processing hook for `Event`, if any.
616 */
617 if constexpr(has_matching_post_processing_hook)
618 {
619 if(running())
620 {
621 const auto processed = impl_.template call_internal_action<false>(*this, context(), event);
622
623 detail::call_matching_event_action<post_processing_hook_ptr_constant_list>
624 (
625 *this,
626 context(),
627 event,
628 processed
629 );
630 }
631 }
632 else
633 {
634 /*
635 Note: We don't need to check if we're running here, as
636 processing the event won't have any effect anyway if the machine
637 is stopped.
638 */
639
640 impl_.template call_internal_action<false>(*this, context(), event);
641 }
642
643 return true;
644 }
645 }
646
647 static constexpr auto pre_processing_hooks = detail::impl_of(conf).pre_processing_hooks;
648 static constexpr auto post_processing_hooks = detail::impl_of(conf).post_processing_hooks;
649
650 using pre_processing_hook_ptr_constant_list = detail::mix_constant_list_t<pre_processing_hooks>;
651 using post_processing_hook_ptr_constant_list = detail::mix_constant_list_t<post_processing_hooks>;
652
653 detail::context_holder
654 <
655 context_type,
656 detail::context_storage::plain,
657 detail::impl_of(conf).context_sig
658 > ctx_holder_;
659
660 impl_type impl_;
