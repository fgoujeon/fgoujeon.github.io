421 template<detail::machine_operation Operation, class Event>
422 void execute_operation(const Event& event)
423 {
424 try
425 {
426 if constexpr(opts(conf).run_to_completion)
427 {
428 if(!executing_operation_) //If call is not recursive
429 {
430 execute_operation_now<Operation>(event);
431 }
432 else
433 {
434 //Enqueue event in case of recursive call
435 enqueue_event_impl<Operation>(event);
436 }
437 }
438 else
439 {
440 execute_one_operation<Operation>(event);
441 }
442 }
443 catch(...)
444 {
445 process_exception(std::current_exception());
446 }
447 }

449 template<detail::machine_operation Operation, class Event>
450 void execute_operation_now(const Event& event)
451 {
452 if constexpr(opts(conf).run_to_completion)

456 execute_one_operation<Operation>(event);

495 template<detail::machine_operation Operation, class Event>
496 void execute_one_operation(const Event& event)
497 {
498 if constexpr(Operation == detail::machine_operation::start)
499 {
500 impl_.call_entry_action(*this, context(), event);
501 }
502 else if constexpr(Operation == detail::machine_operation::stop)
503 {
504 impl_.call_exit_action(*this, context(), event);
505 }
506 else
507 {
508 if constexpr(std::is_same_v<typename option_set_type::fallback_transition_action_tuple_type, detail::tuple<>>)
509 {
510 impl_.call_internal_action(*this, context(), event);
511 }
512 else
513 {
514 auto processed = false;
515 impl_.call_internal_action(*this, context(), event, processed);
516 if(!processed)
517 {
518 detail::call_matching_event_action<fallback_transition_action_ptr_constant_list>
519 (
520 *this,
521 context(),
522 event
523 );
524 }
525 }
526 }
527 }
528
529 static constexpr auto fallback_transition_actions = opts(conf).fallback_transition_actions;
530 using fallback_transition_action_ptr_constant_list = detail::tuple_to_element_ptr_constant_list_t<fallback_transition_actions>;
531
532 detail::context_holder<context_type, opts(conf).context_sig> ctx_holder_;
533 detail::submachine_impl<conf, void, void> impl_;
534 bool executing_operation_ = false;
535 operation_queue_type operation_queue_;
536};
