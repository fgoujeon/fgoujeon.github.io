398 template<detail::machine_operation Operation, class Event>
399 void execute_operation(const Event& event)
400 {
401 try
402 {
403 if constexpr(opts(conf).run_to_completion)
404 {
405 if(!executing_operation_) //If call is not recursive
406 {
407 execute_operation_now<Operation>(event);
408 }
409 else
410 {
411 //Enqueue event in case of recursive call
412 enqueue_event_impl<Operation>(event);
413 }
414 }
415 else
416 {
417 execute_one_operation<Operation>(event);
418 }
419 }
420 catch(...)
421 {
422 process_exception(std::current_exception());
423 }
424 }

426 template<detail::machine_operation Operation, class Event>
427 void execute_operation_now(const Event& event)
428 {
429 if constexpr(opts(conf).run_to_completion)
430 {
431 auto grd = executing_operation_guard{*this};
432
433 execute_one_operation<Operation>(event);
434
435 //Process enqueued events, if any
436 operation_queue_.invoke_and_pop_all(*this);
437 }
438 else
439 {
440 execute_one_operation<Operation>(event);
441 }
442 }

472 template<detail::machine_operation Operation, class Event>
473 void execute_one_operation(const Event& event)
474 {
475 if constexpr(Operation == detail::machine_operation::start)
476 {
477 impl_.call_entry_action(*this, context(), event);
478 }
479 else if constexpr(Operation == detail::machine_operation::stop)
480 {
481 impl_.call_exit_action(*this, context(), event);
482 }
483 else
484 {
485 if constexpr(std::is_same_v<typename option_set_type::post_processing_hook_tuple_type, detail::tuple<>>)
486 {
487 impl_.template call_internal_action<false>(*this, context(), event);
488 }
489 else
490 {
491 auto processed = false;
492 impl_.template call_internal_action<false>(*this, context(), event, processed);
493 detail::call_matching_event_action<post_processing_hook_ptr_constant_list>
494 (
495 *this,
496 context(),
497 event,
498 processed
499 );
500 }
501 }
502 }
503
504 static constexpr auto post_processing_hooks = opts(conf).post_processing_hooks;
505 static constexpr auto path = detail::path{};
506
507 using post_processing_hook_ptr_constant_list = detail::tuple_to_element_ptr_constant_list_t<post_processing_hooks>;
508
509 detail::context_holder<context_type, opts(conf).context_sig> ctx_holder_;
510 detail::composite_state_no_context<&conf, path> impl_;
511 bool executing_operation_ = false;
512 operation_queue_type operation_queue_;
