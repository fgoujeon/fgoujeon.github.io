405 template<detail::machine_operation Operation, class Event>
406 void execute_operation(const Event& event)
407 {
408 try
409 {
410 if constexpr(impl_of(conf).run_to_completion)
411 {
412 if(!executing_operation_) //If call is not recursive
413 {
414 execute_operation_now<Operation>(event);
415 }
416 else
417 {
418 //Enqueue event in case of recursive call
419 enqueue_event_impl<Operation>(event);
420 }
421 }
422 else
423 {
424 execute_one_operation<Operation>(event);
425 }
426 }
427 catch(...)
428 {
429 process_exception(std::current_exception());
430 }
431 }

433 template<detail::machine_operation Operation, class Event>
434 void execute_operation_now(const Event& event)
435 {
436 if constexpr(impl_of(conf).run_to_completion)

440 execute_one_operation<Operation>(event);

442 //Process enqueued events, if any
443 operation_queue_.invoke_and_pop_all(*this);

479 template<detail::machine_operation Operation, class Event>
480 void execute_one_operation(const Event& event)
481 {
482 if constexpr(Operation == detail::machine_operation::start)
483 {
484 impl_.call_entry_action(*this, context(), event);
485 }
486 else if constexpr(Operation == detail::machine_operation::stop)
487 {
488 impl_.exit_to_finals(*this, context(), event);
489 }
490 else
491 {
492 constexpr auto has_matching_pre_processing_hook = detail::tlu::contains_if_v
493 <
494 pre_processing_hook_ptr_constant_list,
495 detail::event_action_traits::for_event<Event>::template has_containing_event_set
496 >;
497
498 constexpr auto has_matching_post_processing_hook = detail::tlu::contains_if_v
499 <
500 post_processing_hook_ptr_constant_list,
501 detail::event_action_traits::for_event<Event>::template has_containing_event_set
502 >;
503
504 //If running, execute pre-processing hook for `Event`, if any.
505 if constexpr(has_matching_pre_processing_hook)
506 {
507 if(running())
508 {
509 detail::call_matching_event_action<pre_processing_hook_ptr_constant_list>
510 (
511 *this,
512 context(),
513 event
514 );
515 }
516 }
517
518 /*
519 If running:
520 - process the event;
521 - execute the post-processing hook for `Event`, if any.
522 */
523 if constexpr(has_matching_post_processing_hook)
524 {
525 if(running())
526 {
527 auto processed = false;
528 impl_.template call_internal_action<false>(*this, context(), event, processed);
529 detail::call_matching_event_action<post_processing_hook_ptr_constant_list>
530 (
531 *this,
532 context(),
533 event,
534 processed
535 );
536 }
537 }
538 else
539 {
540 /*
541 Note: We don't need to check if we're running here, as
542 processing the event won't have any effect anyway if the machine
543 is stopped.
544 */
545
546 impl_.template call_internal_action<false>(*this, context(), event);
547 }
548 }
549 }

552 static constexpr auto post_processing_hooks = impl_of(conf).post_processing_hooks;

555 using pre_processing_hook_ptr_constant_list = detail::tuple_to_element_ptr_constant_list_t<pre_processing_hooks>;
