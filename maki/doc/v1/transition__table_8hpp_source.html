170 template
171 <
172 class SourceStateConf,
173 class EventSet,
174 class TargetStateConfOrNull,
175 class ActionOrNull = null_t,
176 class GuardOrNull = null_t
177 >
178 constexpr auto operator()
179 (
180 const SourceStateConf& source_state_conf,
181 const EventSet& evt_set,
182 const TargetStateConfOrNull& target_state_conf,
183 const ActionOrNull& action = null,
184 const GuardOrNull& guard = null
185 )
186 {
187 //If first transition
188 if constexpr(sizeof...(Transitions) == 0)
189 {
190 static_assert
191 (
192 detail::is_state_conf_v<SourceStateConf>,
193 "Source state of first transition can't be a `maki::state_set`"
194 );
195 }
196
197 static_assert
198 (
199 detail::is_state_conf_v<SourceStateConf> || detail::is_state_set_v<SourceStateConf>,
200 "1st argument must be an instance of `maki::state_conf` or an instance of `maki::state_set`"
201 );
202
203 static_assert
204 (
205 detail::is_event_v<EventSet> || detail::is_event_set_v<EventSet> || detail::is_null_v<EventSet>,
206 "2nd argument must be an instance of `maki::event_t`, an instance of `maki::event_set`, or `maki::null`"
207 );
208
209 static_assert
210 (
211 detail::is_state_conf_v<TargetStateConfOrNull> || detail::is_null_v<TargetStateConfOrNull>,
212 "3rd argument must be an instance of `maki::state_conf` or `maki::null`."
213 );
214
215 static_assert
216 (
217 detail::is_action_v<ActionOrNull> || detail::is_null_v<ActionOrNull>,
218 "4th argument must be an instance of `maki::action` or `maki::null`."
219 );
220
221 static_assert
222 (
223 detail::is_guard_v<GuardOrNull> || detail::is_null_v<GuardOrNull>,
224 "5th argument must be an instance of `maki::guard` or `maki::null`."
225 );
226
227 return detail::make_transition_table
228 (
229 tuple_append
230 (
231 transitions_,
232 detail::transition
233 {
234 detail::store_state_conf(source_state_conf),
235 evt_set,
236 detail::store_state_conf(target_state_conf),
237 detail::to_action(action),
238 detail::to_guard(guard)
239 }
240 )
241 );
242 }
243
244private:
245#ifndef MAKI_DETAIL_DOXYGEN

251#endif

259 detail::tuple<Transitions...> transitions_;
260};
