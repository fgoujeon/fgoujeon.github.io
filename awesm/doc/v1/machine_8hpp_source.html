410 template<detail::machine_operation Operation, class Event>
411 void execute_operation(const Event& event)
412 {
413 try
414 {
415 if constexpr(detail::option_v<conf, detail::option_id::run_to_completion>)
416 {
417 if(!executing_operation_) //If call is not recursive
418 {
419 execute_operation_now<Operation>(event);
420 }
421 else
422 {
423 //Enqueue event in case of recursive call
424 enqueue_event_impl<Operation>(event);
425 }
426 }
427 else
428 {
429 execute_one_operation<Operation>(event);
430 }
431 }
432 catch(...)
433 {
434 process_exception(std::current_exception());
435 }
436 }

438 template<detail::machine_operation Operation, class Event>
439 void execute_operation_now(const Event& event)
440 {
441 if constexpr(detail::option_v<conf, detail::option_id::run_to_completion>)
442 {
443 auto grd = executing_operation_guard{*this};
444
445 execute_one_operation<Operation>(event);
446
447 //Process enqueued events, if any
448 operation_queue_.invoke_and_pop_all(*this);
449 }
450 else
451 {
452 execute_one_operation<Operation>(event);
453 }
454 }

484 template<detail::machine_operation Operation, class Event>
485 void execute_one_operation(const Event& event)
486 {
487 if constexpr(Operation == detail::machine_operation::start)
488 {
489 submachine_.on_entry(event);
490 }
491 else if constexpr(Operation == detail::machine_operation::stop)
492 {
493 submachine_.on_exit(event);
494 }
495 else
496 {
497 if constexpr(detail::option_v<conf, detail::option_id::on_unprocessed>)
498 {
499 auto processed = false;
500 submachine_.on_event(event, processed);
501 if(!processed)
502 {
503 def().on_unprocessed(event);
504 }
505 }
506 else
507 {
508 submachine_.on_event(event);
509 }
510 }
511 }
